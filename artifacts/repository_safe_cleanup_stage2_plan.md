```powershell
$inventory = Import-Csv artifacts\repository_file_inventory_cleanup_plan.csv

  'artifacts/forward_shadow_daily_summary.md',
  'artifacts/forward_shadow_observations.csv',
  'artifacts/v2balanced_historical_evidence_audit.md',
  'artifacts/repository_file_inventory_cleanup_plan.md',
  'artifacts/repository_file_inventory_cleanup_plan.csv',
  'artifacts/repository_safe_cleanup_stage2_plan.md'
)
$artifactArchiveRows = $inventory | Where-Object {
  $_.classification -eq 'ARCHIVE_HISTORICAL_EVIDENCE' `
  -and $_.path -like 'artifacts/*' `
  -and $_.path -notlike 'artifacts/archive/*' `
  -and $active -notcontains $_.path
}
foreach ($row in $artifactArchiveRows) {
  $source = Join-Path (Get-Location) ($row.path -replace '/', '\')
  $target = Join-Path (Get-Location) ('artifacts\archive\' + (Split-Path $row.path -Leaf))
  Move-Item -LiteralPath $source -Destination $target
}
```
